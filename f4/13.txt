#include <iostream>
#include<math.h>

using namespace std;
int main()
{
    setlocale(LC_ALL, "ru"); //подключаем русские буквы
    cout << "вводите кол-во букв ";
    int n; cin >> n; //вводим кол-во букв в строке
    bool b = false; //изначално мы не нашли букву "s" поэтому b=false
    for (size_t i = 0; i < n; i++) //цикл для каждой буквы из строка
    {
        char c; cin >> c; //читаем саму букву
        if (c == 's') //сравниваем с буквой "s" если равен то
        {
            b = true; // то отмечаем что мы нашли букву "s"
            break; //выходим из цикла, так как не имеет смысла дальше проверят
        }
    }
    if (b) //проверям наш флаг, если b=true то
        cout << "в тексте есть буква S" << endl; // выводим текст то что мы нашли букву "s" 
    else //иначе 
        cout << "в тексте нет бука S" << endl; // выводим то что не нашли
    system("pause");
    return 0;
}
