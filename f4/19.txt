#include <iostream>
#include<math.h>
 
using namespace std;
int main()
{
    setlocale(LC_ALL, "ru"); //включаем русские буквы
    cout << "вводите кол-во букв ";
    int n; cin >> n; //вводим кол-во слов в тексте
    cout << "\nвводите слова ";
    string str; cin >> str; //вводим саму слову которое мы посчитаем
    int count = 0; //изначално счетчик равен 0
    for (size_t i = 0; i < n; i++) //создаем цикл для каждой слове в тексте
    {
        string c; cin >> c; //вводим слова текста
        if (c ==str)// проверяем, если слова в тексте равен нашему слову
        {
            count++; //то счетчик плюсуем на 1
        }
    }
    
    cout <<str<< " встречается в тексте "<< count<<" раз" << endl; //выводим счетчик
     
    system("pause");
    return 0;
}
